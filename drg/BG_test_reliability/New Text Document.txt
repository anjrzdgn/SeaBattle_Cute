#include <QApplication>
#include <QWidget>
#include <QLabel>
#include <QPixmap>

int main(int argc, char *argv[]) {
    QApplication app(argc, argv);

    QWidget mainWindow;
    mainWindow.setGeometry(100, 100, 800, 600); // Set window size

    QLabel backgroundImageLabel(&mainWindow);
    QPixmap backgroundImage("path/to/your/image.jpg"); // Load your image
    backgroundImageLabel.setPixmap(backgroundImage);

    QLabel textLabel("Hello, World!", &mainWindow);
    textLabel.setGeometry(50, 50, 200, 30); 

    // Show the window
    mainWindow.show();

    return app.exec();
}
